```typescript
export async function generateMetadata({ params }: { params: Promise<{ slug: string }> }): Promise<Metadata> {
  const { slug } = await params;
  const caseStudy = await getCaseStudyBySlug(slug);
  return {
    title: caseStudy.title,
    description: caseStudy.excerpt,
  };
}
```
Note: In Next.js 15, params is a Promise — must await it.

**Page component — Server Component:**
```typescript
export default async function CaseStudyPage({ params }: { params: Promise<{ slug: string }> }) {
  const { slug } = await params;
  const caseStudy = await getCaseStudyBySlug(slug);

  // Dynamic import for MDX content
  const MDXContent = (await import(`@/content/portfolio/${slug}.mdx`)).default;

  // Get related case studies for up-next navigation
  const allStudies = await getAllCaseStudies();
  const relatedStudies = caseStudy.relatedSlugs
    .map(s => allStudies.find(cs => cs.slug === s))
    .filter(Boolean);

  if (caseStudy.tier === "A") {
    return <TierALayout caseStudy={caseStudy} related={relatedStudies}><MDXContent /></TierALayout>;
  }
  return <TierBLayout caseStudy={caseStudy} related={relatedStudies}><MDXContent /></TierBLayout>;
}
```
